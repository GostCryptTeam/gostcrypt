34 {
35  char buf[PATH_MAX], *path = NULL;
36  dev_t dev = 0;
37

40  /*
41  * Read from /dev
42  */

44 
45  if (stat(name, &st) == 0)
46  {
47  dev = st.st_rdev;
48  }
49  else
50  {
51  name += 5; /* unaccesible, or not node in /dev */
52  }
53  }
54 
55  if (!dev && parent && strncmp("dm-", name, 3))
56  {
57  /*
58  * Create path to /sys/block/<parent>/<name>/dev
59  */
60  int len = snprintf(buf, sizeof(buf),
61  _PATH_SYS_BLOCK "/%s/%s/dev", parent, name);
62  if (len < 0 || (size_t) len + 1 > sizeof(buf))
63  {
64  return 0;
65  }
66  path = buf;
67 
68  }
69  else if (!dev)
70  {
71  /*
72  * Create path to /sys/block/<name>/dev
73  */
74  int len = snprintf(buf, sizeof(buf),
75  _PATH_SYS_BLOCK "/%s/dev", name);
76  if (len < 0 || (size_t) len + 1 > sizeof(buf))
77  {
78  return 0;
79  }
80  path = buf;
81  }
82

84  {
85  /*
86  * read devno from sysfs
87  */
88  FILE* f;

102  }
