48 void VolumeHeader::Create(BufferPtr& headerBuffer, VolumeHeaderCreationOptions& options)
49 {
50  if (options.DataKey.size() != options.EA->GetKeySize() * 2)
51  {
52  throw InvalidParameterException("options.DataKey.size()",
53  "Given Key size different from the encryption algorithm key size");
54  }
55  if (options.Salt.size() != GetSaltSize())
56  {
57  throw InvalidParameterException("options.Salt.size()", "Given salt size incorrect");
58  }
59 
60  headerBuffer.erase();
61 
62  HeaderVersion = CurrentHeaderVersion;
63  RequiredMinProgramVersion = CurrentRequiredMinProgramVersion;
64 
65  DataAreaKey.erase();
66  DataAreaKey.copyFrom(options.DataKey);
67 
68  VolumeCreationTime = 0;
69  HiddenVolumeDataSize = (options.Type == VolumeType::Hidden ? options.VolumeDataSize :
70  0); // to know if the volume is hidden or not. useless since the layout already contains this information.
71  VolumeDataSize = options.VolumeDataSize;
72 
73  EncryptedAreaStart = options.VolumeDataStart;
74  EncryptedAreaLength = options.VolumeDataSize;
75 
76  SectorSize = options.SectorSize;
77 
78  if (SectorSize < GST_MIN_VOLUME_SECTOR_SIZE
79  || SectorSize > GST_MAX_VOLUME_SECTOR_SIZE
80  || SectorSize % ENCRYPTION_DATA_UNIT_SIZE != 0)
81  {
82  throw InvalidParameterException("SectorSize",
83  "The sector size should be greater than " + QString::number(GST_MIN_VOLUME_SECTOR_SIZE) +
84  ", lower than " + QString::number(GST_MAX_VOLUME_SECTOR_SIZE) + " and multiple of " +
85  QString::number(ENCRYPTION_DATA_UNIT_SIZE) + ".");
86  }
87 
88  EA = options.EA;
89 
90  EncryptNew(headerBuffer, options.Salt, options.HeaderKey, options.Hash);

93 bool VolumeHeader::Decrypt(const BufferPtr& encryptedData, const VolumePassword& password,
94  const VolumeHashList& keyDerivationFunctions, const EncryptionAlgorithmList& encryptionAlgorithms)
95 {
96  if (password.Size() < 1)
97  {
98  throw InvalidParameterException("Password", "Empty password");
99  }
100 
101  const BufferPtr salt(encryptedData.getRange(SaltOffset, SaltSize));
102  SecureBuffer header(EncryptedHeaderDataSize);
103  SecureBuffer headerKey(GetLargestSerializedKeySize());
104 
105  for (QSharedPointer <VolumeHash> derivationfunction : keyDerivationFunctions)
106  {
107  derivationfunction->HMAC_DeriveKey(headerKey, password, salt);
108 
109  for (QSharedPointer <EncryptionAlgorithm> ea : encryptionAlgorithms)
110  {
111  ea->SetKey(headerKey.getRange(0, ea->GetKeySize()));
112  ea->GetMode()->SetKey(headerKey.getRange(ea->GetKeySize(), ea->GetKeySize()));
113 
114  header.copyFrom(encryptedData.getRange(EncryptedHeaderDataOffset, EncryptedHeaderDataSize));
115  ea->Decrypt(header);
116 
117  if (Deserialize(header, ea))
118  {
119  this->EA = ea;
120  this->volumeHash = derivationfunction;
121  return true;
122  }
123  }
124  }
125 
126  return false;
127 }

130 bool VolumeHeader::Deserialize(const BufferPtr& header, QSharedPointer <EncryptionAlgorithm>& ea)
131 {
132 
133  if (header.size() != EncryptedHeaderDataSize)
134  {
135  throw InvalidParameterException("header.size()", "");
136  }
137 
138  BufferPtr magicNumber((const quint8*)"TRUE", 4);
139  if (!magicNumber.isDataEqual(header.getRange(0, 4)))
140  {
141  return false;
142  }
143 
144  /* Vérification de la version du header */
145  size_t offset = 4;
146  HeaderVersion = DeserializeEntry <quint16> (header, offset);
147 
148  if (HeaderVersion < MinAllowedHeaderVersion)
149  {
150  throw VolumeVersionNotCompatibleException("Volume version is " + QString::number(
151  MinAllowedHeaderVersion, 16) + " and is no longer supported");
152  }
153 
154  if (HeaderVersion > CurrentHeaderVersion)
155  {
156  throw VolumeVersionNotCompatibleException("Volume version is " + QString::number(HeaderVersion,
157  16) + " but this GostCrypt version support only the version " + QString::number(
158  CurrentHeaderVersion, 16) + " and below.");
159  }
160 
161  if (HeaderVersion >= 4
162  && Crc32::ProcessBuffer(header.getRange(0, GST_HEADER_OFFSET_HEADER_CRC - GST_HEADER_OFFSET_MAGIC))
163  != DeserializeEntryAt <quint32> (header, GST_HEADER_OFFSET_HEADER_CRC - GST_HEADER_OFFSET_MAGIC))
164  {
165  return false;
166  }
167 
168  RequiredMinProgramVersion = DeserializeEntry <quint16> (header, offset);
169 
170 
171  if (RequiredMinProgramVersion > VERSION_NUM)
172  {
173  throw VolumeVersionNotCompatibleException("Volume version too young");
174  }
175 
176  VolumeKeyAreaCrc32 = DeserializeEntry <quint32> (header, offset);
177  VolumeCreationTime = DeserializeEntry <quint64> (header, offset);
178  HeaderCreationTime = DeserializeEntry <quint64> (header, offset);
179  HiddenVolumeDataSize = DeserializeEntry <quint64> (header, offset);
180  mVolumeType = (HiddenVolumeDataSize != 0 ? VolumeType::Hidden : VolumeType::Normal);
181  VolumeDataSize = DeserializeEntry <quint64> (header, offset);
182  EncryptedAreaStart = DeserializeEntry <quint64> (header, offset);
183  EncryptedAreaLength = DeserializeEntry <quint64> (header, offset);
184  Flags = DeserializeEntry <quint32> (header, offset);
185 
186  SectorSize = DeserializeEntry <quint32> (header, offset);
187  if (HeaderVersion < 5)
188  {
189  SectorSize = GST_SECTOR_SIZE_LEGACY;
190  }
191 
192  if (SectorSize < GST_MIN_VOLUME_SECTOR_SIZE
193  || SectorSize > GST_MAX_VOLUME_SECTOR_SIZE
194  || SectorSize % ENCRYPTION_DATA_UNIT_SIZE != 0)
195  {
196  throw InvalidParameterException("SectorSize",
197  "The sector size should be greater than " + QString::number(GST_MIN_VOLUME_SECTOR_SIZE) +
198  ", lower than " + QString::number(GST_MAX_VOLUME_SECTOR_SIZE) + " and multiple of " +
199  QString::number(ENCRYPTION_DATA_UNIT_SIZE) + ".");
200  }
201 
202 #if !(defined (GST_WINDOWS) || defined (GST_LINUX))
203  if (SectorSize != GST_SECTOR_SIZE_LEGACY)
204  {
205  throw UnsupportedSectorSize(SRC_POS);
206  }
207 #endif
208 
209  offset = DataAreaKeyOffset;
210 
211  if (VolumeKeyAreaCrc32 != Crc32::ProcessBuffer(header.getRange(offset, DataKeyAreaMaxSize)))
212  {
213  return false;
214  }
215 
216  DataAreaKey.copyFrom(header.getRange(offset, DataKeyAreaMaxSize));
217 
218  //Reset keys for use on the whole volume
219  ea->SetKey(header.getRange(offset, ea->GetKeySize()));
220  ea->GetMode()->SetKey(header.getRange(offset + ea->GetKeySize(), ea->GetKeySize()));
221 
222  return true;
223 }

226 template <typename T>
227 T VolumeHeader::DeserializeEntry(const BufferPtr& header, size_t& offset)
228 {
229  offset += sizeof(T);
230 
231  if (offset > header.size())
232  {
233  throw InvalidParameterException("offset",
234  "Trying to deserialize header entry after the end of the header");
235  }
236 
237  return qToBigEndian(*reinterpret_cast<const T*>(header.get() + offset - sizeof(T)));
238 }

252 void VolumeHeader::EncryptNew(BufferPtr& newHeaderBuffer, const BufferPtr& newSalt,
253  const BufferPtr& newHeaderKey, QSharedPointer <VolumeHash> newVolumeHash)
254 {
255  if (newHeaderBuffer.size() != HeaderSize)
256  {
257  throw InvalidParameterException("ewHeaderBuffer.size()", "Incorrect new header buffer size");
258  }
259  if (newSalt.size() != SaltSize)
260  {
261  throw InvalidParameterException("newSalt.size()", "Incorrect new salt size");
262  }
263 
264  QSharedPointer <EncryptionAlgorithm> ea = EA->GetNew();
265 
266  ea->GetMode()->SetKey(newHeaderKey.getRange(ea->GetKeySize(), ea->GetKeySize()));
267  ea->SetKey(newHeaderKey.getRange(0, ea->GetKeySize()));
268 
269  newHeaderBuffer.copyFrom(newSalt);
270 
271  BufferPtr headerData = newHeaderBuffer.getRange(EncryptedHeaderDataOffset, EncryptedHeaderDataSize);
272  Serialize(headerData);
273  ea->Encrypt(headerData);
274 
275  if (newVolumeHash)
276  {
277  this->volumeHash = newVolumeHash;
278  }
279 }

281 size_t VolumeHeader::GetLargestSerializedKeySize()
282 {
283  size_t largestKey = EncryptionAlgorithm::GetLargestKeySize(
284  EncryptionAlgorithm::GetAvailableAlgorithms());
285 
286  // XTS mode requires the same key size as the encryption algorithm.

291  }

296 void VolumeHeader::Serialize(BufferPtr& header) const
297 {
298  if (header.size() != EncryptedHeaderDataSize)
299  {
300  throw InvalidParameterException("header.size()", "Incorrect header buffer size");
301  }
302 
303  header.erase();
304 
305  header[0] = 'T';
306  header[1] = 'R';
307  header[2] = 'U';
308  header[3] = 'E';
309  size_t offset = 4;
310 
311  header.getRange(DataAreaKeyOffset, DataAreaKey.size()).copyFrom(DataAreaKey);
312 
313  quint16 headerVersion = CurrentHeaderVersion;
314  SerializeEntry(headerVersion, header, offset);
315  SerializeEntry(RequiredMinProgramVersion, header, offset);
316  SerializeEntry(Crc32::ProcessBuffer(header.getRange(DataAreaKeyOffset, DataKeyAreaMaxSize)), header,
317  offset);
318 
319  quint64 reserved64 = 0;
320  SerializeEntry(reserved64, header, offset);
321  SerializeEntry(reserved64, header, offset);
322 
323  SerializeEntry(HiddenVolumeDataSize, header, offset);
324  SerializeEntry(VolumeDataSize, header, offset);
325  SerializeEntry(EncryptedAreaStart, header, offset);
326  SerializeEntry(EncryptedAreaLength, header, offset);
327  SerializeEntry(Flags, header, offset);
328 
329  if (SectorSize < GST_MIN_VOLUME_SECTOR_SIZE
330  || SectorSize > GST_MAX_VOLUME_SECTOR_SIZE
331  || SectorSize % ENCRYPTION_DATA_UNIT_SIZE != 0)
332  {
333  throw InvalidParameterException("SectorSize",
334  "The sector size should be greater than " + QString::number(GST_MIN_VOLUME_SECTOR_SIZE) +
335  ", lower than " + QString::number(GST_MAX_VOLUME_SECTOR_SIZE) + " and multiple of " +
336  QString::number(ENCRYPTION_DATA_UNIT_SIZE) + ".");
337  }
338 
339  SerializeEntry(SectorSize, header, offset);
340 
341  offset = GST_HEADER_OFFSET_HEADER_CRC - GST_HEADER_OFFSET_MAGIC;
342  SerializeEntry(Crc32::ProcessBuffer(header.getRange(0,
343  GST_HEADER_OFFSET_HEADER_CRC - GST_HEADER_OFFSET_MAGIC)), header, offset);
344 }
