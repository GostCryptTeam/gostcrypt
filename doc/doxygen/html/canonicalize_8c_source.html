20 /*
21  * Converts private "dm-N" names to "/dev/mapper/<name>"
22  *
23  * Since 2.6.29 (patch 784aae735d9b0bba3f8b9faef4c8b30df3bf0128) kernel sysfs
24  * provides the real DM device names in /sys/block/<ptname>/dm/name
25  */
26 char* canonicalize_dm_name(const char* ptname)
27 {
28  FILE* f;
29  size_t sz;
30  char path[256], name[256], *res = NULL;
31 
32  if (!ptname || !*ptname)
33  {
34  return NULL;
35  }
36 
37  snprintf(path, sizeof(path), "/sys/block/%s/dm/name", ptname);
38  if (!(f = fopen(path, "r" UL_CLOEXECSTR)))
39  {
40  return NULL;
41  }
42 
43  /* read "<name>\n" from sysfs */
44  if (fgets(name, sizeof(name), f) && (sz = strlen(name)) > 1)
45  {
46  name[sz - 1] = '\0';
47  snprintf(path, sizeof(path), "/dev/mapper/%s", name);
48 
49  if (access(path, F_OK) == 0)
50  {
51  res = strdup(path);
52  }
53  }
54  fclose(f);
55  return res;
56 }

59 {

61 
62  if (!path || !*path)
63  {

66

72

74  if (p && strncmp(p, "/dm-", 4) == 0 && isdigit(*(p + 4)))
75  {
76  char* dm = canonicalize_dm_name(p + 1);
