125 void RandomGenerator::GetData(BufferPtr& buffer, bool fast)
126 {
127  if (!Running)
128  {
129  throw RandomNumberGeneratorNotRunningException();
130  }
131 
132  if (buffer.size() > PoolSize)
133  {
134  throw InvalidParameterException("buffer.size()", "buffer.size() > PoolSize");
135  }
136 
137  QMutexLocker lock(&AccessMutex);
138 
139  // Poll system for data
140  AddSystemDataToPool(fast);
141  HashMixPool();
142 
143  // Transfer bytes from pool to output buffer
144  for (size_t i = 0; i < buffer.size(); ++i)
145  {
146  buffer[i] += Pool[ReadOffset++];
147 
148  if (ReadOffset >= PoolSize)
149  {
150  ReadOffset = 0;
151  }
152  }
153 
154  // Invert and mix the pool
155  for (size_t i = 0; i < Pool.size(); ++i)
156  {
157  Pool[i] = ~Pool[i];
158  }
159 
160  AddSystemDataToPool(true);
161  HashMixPool();
162 
163  // XOR the current pool content into the output buffer to prevent pool state leaks
164  for (size_t i = 0; i < buffer.size(); ++i)
165  {
166  buffer[i] ^= Pool[ReadOffset++];
167 
168  if (ReadOffset >= PoolSize)
169  {
170  ReadOffset = 0;
171  }
172  }
173 }

175 void RandomGenerator::HashMixPool()
176 {
177  BytesAddedSincePoolHashMix = 0;
178 
179  for (size_t poolPos = 0; poolPos < Pool.size();)
180  {
181  // Compute the message digest of the entire pool using the selected hash function
182  SecureBuffer digest(PoolHash->GetDigestSize());
183  PoolHash->ProcessData(Pool);
184  PoolHash->GetDigest(digest);
185 
186  // Add the message digest to the pool
187  for (size_t digestPos = 0; digestPos < digest.size() && poolPos < Pool.size(); ++digestPos)
188  {
189  Pool[poolPos++] += digest[digestPos];
190  }
191  }
192 }
