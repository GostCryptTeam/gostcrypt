67 void Volume::Open(const QFileInfo volumePath,
68  bool preserveTimestamps,
69  QSharedPointer <VolumePassword> password,
70  QSharedPointer <KeyfileList> keyfiles,
71  VolumeProtection::Enum protection,
72  QSharedPointer <VolumePassword> protectionPassword, // in case you wqnt to open it as
73  QSharedPointer <KeyfileList> protectionKeyfiles,
74  VolumeType::Enum volumeType,
75  bool useBackupHeaders)
76 {
77  QSharedPointer<VolumeFile> volumeFile = QSharedPointer<VolumeFile>(new VolumeFile());
78 
79  if (protection == VolumeProtection::HiddenVolumeReadOnly)
80  {
81  // we first open the hidden volume to get its size
82  try
83  {
84  Volume protectedVolume;
85 
86  protectedVolume.Open(volumePath,
87  preserveTimestamps,
88  protectionPassword, protectionKeyfiles,
89  VolumeProtection::ReadOnly,
90  QSharedPointer <VolumePassword> (), QSharedPointer <KeyfileList> (),
91  VolumeType::Hidden,
92  useBackupHeaders);
93  if (protectedVolume.GetType() != VolumeType::Hidden)
94  {
95  throw InvalidParameterException("password/keyfiles",
96  "The given password/keyfiles to open the inner hidden volume instance are the ones of the outer volume instance");
97  }
98 
99  ProtectedRangeStart = protectedVolume.VolumeDataOffset;
100  ProtectedRangeEnd = protectedVolume.VolumeDataOffset + protectedVolume.VolumeDataSize;
101  }
102  catch (PasswordOrKeyfilesIncorrect&)
103  {
104  throw ProtectionPasswordOrKeyfilesIncorrectException(volumePath);
105  }
106  }
107 
108  // We First open the file (or device) we want to decrypt
109  volumeFile->Open(volumePath, (protection == VolumeProtection::ReadOnly), preserveTimestamps);
110 
111  Protection = protection;
112  this->volumeFile = volumeFile; // why not directly use the volumeFile from the object ?
113 
114  try
115  {
116  VolumeHostSize = this->volumeFile->Length();
117  QSharedPointer <VolumePassword> passwordKey = Keyfile::ApplyListToPassword(keyfiles, password);
118  // Test volume layouts
119  for (QSharedPointer <VolumeLayout> layout : VolumeLayout::GetAvailableLayouts(volumeType))
120  {
121 
122  if (useBackupHeaders && !layout->HasBackupHeader())
123  {
124  continue;
125  }
126 
127  SecureBuffer headerBuffer(layout->GetHeaderSize());
128 
129  int headerOffset = useBackupHeaders ? layout->GetBackupHeaderOffset() : layout->GetHeaderOffset();
130 
131  if (headerOffset >= 0)
132  {
133  this->volumeFile->SeekAt(headerOffset);
134  }
135  else
136  {
137  this->volumeFile->SeekEnd(headerOffset);
138  }
139 
140  if (this->volumeFile->Read(headerBuffer) != layout->GetHeaderSize())
141  {
142  continue;
143  }
144 
145  EncryptionAlgorithmList layoutEncryptionAlgorithms = layout->GetSupportedEncryptionAlgorithms();
146 
147  QSharedPointer <VolumeHeader> header = layout->GetHeader();
148 
149  if (header->Decrypt(headerBuffer, *passwordKey, layout->GetSupportedKeyDerivationFunctions(),
150  layoutEncryptionAlgorithms))
151  {
152  // Header decrypted
153 
154  Type = layout->GetType();
155  SectorSize = header->GetSectorSize();
156 
157  VolumeDataOffset = layout->GetDataOffset(VolumeHostSize);
158  VolumeDataSize = layout->GetDataSize(VolumeHostSize);
159 
160  Header = header;
161  Layout = layout;
162  EA = header->GetEncryptionAlgorithm();
163 
164  // Volume protection
165  if (Protection == VolumeProtection::HiddenVolumeReadOnly)
166  {
167  if (Type == VolumeType::Hidden)
168  {
169  throw PasswordOrKeyfilesIncorrectException(
170  volumePath); // the password of the inner volume was put instead of the one of the outer volume.
171  }
172  // protectedrangestart and protectedrangeend were set before
173  }
174  return;
175  }
176  }
177  throw PasswordOrKeyfilesIncorrectException(volumePath);
178  }
179  catch (...)
180  {
181  Close();
182  throw; //rethrow
183  }
184 }

186 void Volume::ReadSectors(BufferPtr& buffer, quint64 byteOffset)
187 {
188  //if_debug (ValidateState ());
189 
190  quint64 length = buffer.size();
191  quint64 hostOffset = VolumeDataOffset + byteOffset;
192 
193  if (length % SectorSize != 0 || byteOffset % SectorSize != 0)
194  {
195  throw InvalidParameterException("length or offset", "length or offset not aligned with sector");
196  }
197 
198  if (this->volumeFile->ReadAt(buffer, hostOffset) != length)
199  {
200  throw VolumeCorruptedException();
201  }
202 
203  EA->DecryptSectors(buffer, hostOffset / SectorSize, length / SectorSize, SectorSize);
204 
205  TotalDataRead += length;

236 void Volume::WriteSectors(const BufferPtr& buffer, quint64 byteOffset)
237 {
238  quint64 length = buffer.size();
239  quint64 hostOffset = VolumeDataOffset + byteOffset;
240 
241  if (length % SectorSize != 0
242  || byteOffset % SectorSize != 0)
243  {
244  throw InvalidParameterException("length or offset", "length or offset not aligned with sector");
245  }
246 
247  if (byteOffset > VolumeDataSize)
248  {
249  throw InvalidParameterException("byteOffset", "Trying to read after the end of the volume file");
250  }
251 
252  if (Protection == VolumeProtection::ReadOnly)
253  {
254  throw VolumeReadOnlyException();
255  }
256  if (HiddenVolumeProtectionTriggered)//TODO why always throwing when the protection is triggered ?
257  {
258  throw VolumeProtectedException();
259  }
260 
261  if (Protection == VolumeProtection::HiddenVolumeReadOnly)
262  {
263  CheckProtectedRange(hostOffset, length);
264  }
265 
266  SecureBuffer encBuf(buffer.size());
267  encBuf.copyFrom(buffer);
268 
269  EA->EncryptSectors(encBuf, hostOffset / SectorSize, length / SectorSize, SectorSize);
270  this->volumeFile->WriteAt(encBuf, hostOffset);
271 
272  TotalDataWritten += length;
273 
274  quint64 writeEndOffset = byteOffset + buffer.size();
275  if (writeEndOffset > TopWriteOffset)
276  {
277  TopWriteOffset = writeEndOffset;
278  }
279 }
