21 void EncryptionThreadPool::DoWork(WorkType::Enum type, const EncryptionMode* encryptionMode,
22  quint8* data, quint64 startUnitNo, quint64 unitCount, size_t sectorSize)
23 {
24  size_t fragmentCount;
25  size_t unitsPerFragment;
26  size_t remainder;
27 
28  quint8* fragmentData;
29  quint64 fragmentStartUnitNo;
30 
31  WorkItem* firstFragmentWorkItem;
32 
33  if (unitCount == 0)
34  {
35  return;
36  }
37 
38  if (!ThreadPoolRunning || unitCount == 1)
39  {
40  switch (type)
41  {
42  case WorkType::DecryptDataUnits:
43  encryptionMode->DecryptSectorsCurrentThread(data, startUnitNo, unitCount, sectorSize);
44  break;
45 
46  case WorkType::EncryptDataUnits:
47  encryptionMode->EncryptSectorsCurrentThread(data, startUnitNo, unitCount, sectorSize);
48  break;
49 
50  default:
51  throw InvalidParameterException("WorkType", "Unknown WorkType");
52  }
53 
54  return;
55  }
56 
57  if (unitCount <= ThreadCount)
58  {
59  fragmentCount = (size_t) unitCount;
60  unitsPerFragment = 1;
61  remainder = 0;
62  }
63  else
64  {
65  fragmentCount = ThreadCount;
66  unitsPerFragment = (size_t) unitCount / ThreadCount;
67  remainder = (size_t) unitCount % ThreadCount;
68 
69  if (remainder > 0)
70  {
71  ++unitsPerFragment;
72  }
73  }
74 
75  fragmentData = data;
76  fragmentStartUnitNo = startUnitNo;
77 
78  {
79  QMutexLocker lock(&EnqueueMutex);
80  firstFragmentWorkItem = &WorkItemQueue[EnqueuePosition];
81 
82  while (firstFragmentWorkItem->State != WorkItem::State::Free)
83  {
84  WorkItemCompletedEvent.wait(&WorkItemCompletedEventMutex);
85  }
86 
87  firstFragmentWorkItem->OutstandingFragmentCount = fragmentCount;
88  firstFragmentWorkItem->ItemException.reset();
89 
90  while (fragmentCount-- > 0)
91  {
92  WorkItem* workItem = &WorkItemQueue[EnqueuePosition++];
93 
94  if (EnqueuePosition >= QueueSize)
95  {
96  EnqueuePosition = 0;
97  }
98 
99  while (workItem->State != WorkItem::State::Free)
100  {
101  WorkItemCompletedEvent.wait(&WorkItemCompletedEventMutex);
102  }
103 
104  workItem->Type = type;
105  workItem->FirstFragment = firstFragmentWorkItem;
106 
107  workItem->Encryption.Mode = encryptionMode;
108  workItem->Encryption.Data = fragmentData;
109  workItem->Encryption.UnitCount = unitsPerFragment;
110  workItem->Encryption.StartUnitNo = fragmentStartUnitNo;
111  workItem->Encryption.SectorSize = sectorSize;
112 
113  fragmentData += unitsPerFragment * ENCRYPTION_DATA_UNIT_SIZE;
114  fragmentStartUnitNo += unitsPerFragment;
115 
116  if (remainder > 0 && --remainder == 0)
117  {
118  --unitsPerFragment;
119  }
120 
121  workItem->State = WorkItem::State::Ready;
122 
123  EncryptionThreadPool::WorkItemReadyEventMutex.unlock();
124  WorkItemReadyEvent.wakeOne();
125  }
126  firstFragmentWorkItem->ItemCompletedEventMutex.lock();
127  }
128 
129  firstFragmentWorkItem->ItemCompletedEvent.wait(&firstFragmentWorkItem->ItemCompletedEventMutex);
130 
131  QSharedPointer <GostCryptException> itemException;
132  if (!firstFragmentWorkItem->ItemException.isNull())
133  {
134  itemException = firstFragmentWorkItem->ItemException;
135  }
136 
137  firstFragmentWorkItem->State = WorkItem::State::Free;
138  firstFragmentWorkItem->ItemCompletedEventMutex.unlock();
139 
140  WorkItemCompletedEvent.wakeOne();
141 
142  if (!itemException.isNull())
143  {
144  itemException->raise();
145  }

148 void EncryptionThreadPool::Start()
149 {
150  if (ThreadPoolRunning)
151  {
152  return;
153  }
154 
155  int cpuCount;
156 
157  cpuCount = QThread::idealThreadCount();
158  if (cpuCount == -1)
159  {
160  cpuCount = 1;
161  }
162 
163 
164  if (cpuCount < 2)
165  {
166  return;
167  }
168 
169  if (cpuCount > MaxThreadCount)
170  {
171  cpuCount = MaxThreadCount;
172  }
173 
174  StopPending = false;
175  DequeuePosition = 0;
176  EnqueuePosition = 0;
177 
178  for (size_t i = 0; i < sizeof(WorkItemQueue) / sizeof(WorkItemQueue[0]); ++i)
179  {
180  WorkItemQueue[i].State = WorkItem::State::Free;
181  }
182 
183  try
184  {
185  for (ThreadCount = 0; ThreadCount < cpuCount; ++ThreadCount)
186  {
187  QSharedPointer<EncryptionThread> thread(new EncryptionThread());
188  thread->start();
189  RunningThreads.append(thread);
190  }
191  }
192  catch (...)
193  {
194  try
195  {
196  ThreadPoolRunning = true;
197  Stop();
198  }
199  catch (...) { }
200 
201  throw; //rethrow
202  }
203  WorkItemCompletedEventMutex.lock();
204  WorkItemReadyEventMutex.lock();
205 
206  ThreadPoolRunning = true;
207 }

209 void EncryptionThreadPool::Stop()
210 {
211  if (!ThreadPoolRunning)
212  {
213  return;
214  }
215 
216  StopPending = true;
217  WorkItemReadyEvent.wakeAll();
218 
219  for (const QSharedPointer<EncryptionThread> thread : RunningThreads)
220  {
221  thread->wait();
222  }
223  WorkItemCompletedEventMutex.unlock();
224  WorkItemReadyEventMutex.unlock();
225  ThreadCount = 0;
226  ThreadPoolRunning = false;

231 void EncryptionThread::run()
232 {
233  try
234  {
235  while (!EncryptionThreadPool::StopPending)
236  {
237  EncryptionThreadPool::WorkItem* workItem;
238  {
239  QMutexLocker lock(&EncryptionThreadPool::DequeueMutex);
240 
241  workItem = &EncryptionThreadPool::WorkItemQueue[EncryptionThreadPool::DequeuePosition++];
242 
243  if (EncryptionThreadPool::DequeuePosition >= EncryptionThreadPool::QueueSize)
244  {
245  EncryptionThreadPool::DequeuePosition = 0;
246  }
247 
248  // Waiting for current workItem to be Ready (there is work to do)
249  while (!EncryptionThreadPool::StopPending &&
250  workItem->State != EncryptionThreadPool::WorkItem::State::Ready)
251  {
252  EncryptionThreadPool::WorkItemReadyEvent.wait(
253  &EncryptionThreadPool::WorkItemReadyEventMutex);//, 100);
254  }
255 
256  workItem->State = EncryptionThreadPool::WorkItem::State::Busy;
257  }
258 
259  if (EncryptionThreadPool::StopPending)
260  {
261  break;
262  }
263 
264  try
265  {
266  switch (workItem->Type)
267  {
268  case EncryptionThreadPool::WorkType::DecryptDataUnits:
269  workItem->Encryption.Mode->DecryptSectorsCurrentThread(workItem->Encryption.Data,
270  workItem->Encryption.StartUnitNo, workItem->Encryption.UnitCount, workItem->Encryption.SectorSize);
271  break;
272 
273  case EncryptionThreadPool::WorkType::EncryptDataUnits:
274  workItem->Encryption.Mode->EncryptSectorsCurrentThread(workItem->Encryption.Data,
275  workItem->Encryption.StartUnitNo, workItem->Encryption.UnitCount, workItem->Encryption.SectorSize);
276  break;
277 
278  default:
279  throw InvalidParameterException("WorkType", "Unknown WorkType");
280  }
281  }
282  catch (GostCryptException& e)
283  {
284  workItem->FirstFragment->ItemException.reset(e.clone());
285  }
286  catch (std::exception& e)
287  {
288  workItem->FirstFragment->ItemException.reset(new ExternalExceptionException(e));
289  }
290  catch (...)
291  {
292  workItem->FirstFragment->ItemException.reset(new UnknowExceptionException());
293  }
294 
295  if (workItem != workItem->FirstFragment)
296  {
297  workItem->State = EncryptionThreadPool::WorkItem::State::Free;
298  EncryptionThreadPool::WorkItemCompletedEvent.wakeOne();
299  }
300 
301  if (!workItem->FirstFragment->OutstandingFragmentCount.deref())
302  {
303  workItem->FirstFragment->ItemCompletedEvent.wakeOne();
304  }
305  }
306  }
307  catch (std::exception& e)
308  {
309  // SystemLog::WriteException (e); TODO
310  }
311  catch (...)
312  {
313  //SystemLog::WriteException (UnknownException (SRC_POS));
314  }
315 }
