20 void Keyfile::Apply(BufferPtr& pool) const
21 {
22  QFile file(Path.absoluteFilePath());
23  Crc32 crc32;
24  size_t poolPos = 0;
25  quint64 totalLength = 0;
26  qint64 readLength;
27  SecureBuffer keyfileBuf(FILE_OPTIMAL_READ_SIZE);
28 
29  file.open(QIODevice::ReadOnly);
30 
31  while ((readLength = file.read(reinterpret_cast<char*>(keyfileBuf.get()), keyfileBuf.size())))
32  {
33  if (readLength == -1)
34  {
35  throw FailedReadFileException(Path);
36  }
37  for (qint64 i = 0; i < readLength; i++)
38  {
39  quint32 crc = crc32.Process(keyfileBuf[i]);
40 
41  pool[poolPos++] += (quint8)(crc >> 24);
42  pool[poolPos++] += (quint8)(crc >> 16);
43  pool[poolPos++] += (quint8)(crc >> 8);
44  pool[poolPos++] += (quint8) crc;
45 
46  if (poolPos >= pool.size())
47  {
48  poolPos = 0;
49  }
50 
51  if (++totalLength >= MaxProcessedLength)
52  {
53  return;
54  }
55  }
56  }
57 
58  file.close();
59 }
