338  containerSize); // unlike truecrypt, we let the user set its own size
339 
340  if (options.Type == Volume::VolumeType::Hidden)
341  {
342  options.VolumeDataStart = containerSize - layout->GetHeaderSize() * 2 - options.VolumeDataSize;
343  }

347  }

352  }
353 
354

356  GostCrypt::SecureBuffer salt; // salt to encrypt the header.
357  GostCrypt::SecureBuffer headerkey; // fusion of the password and the keyfiles, derived
358 
359  SecureBuffer headerBuffer(layout->GetHeaderSize()); // our header we'll write to the container
360 
361  // Master data key
362  masterkey.allocate(options.EA->GetKeySize() * 2);
363  RandomGenerator::GetData(masterkey);
364  options.DataKey = masterkey;
365 
366  // PKCS5 salt
367  salt.allocate(Volume::VolumeHeader::GetSaltSize());
368  RandomGenerator::GetData(salt);
369  options.Salt = salt;
370

372  headerkey.allocate(Volume::VolumeHeader::GetLargestSerializedKeySize());
373  QSharedPointer <Volume::KeyfileList> keyfiles;
374  for (QFileInfo keyfile : params->keyfiles)

376  keyfiles->append(QSharedPointer<Volume::Keyfile>(new Volume::Keyfile(keyfile)));
377  }

387  }

389  password);

391  options.HeaderKey = headerkey;
392

394 
395  // Write The Header

408  }
409  file.write((char*)headerBuffer.get(), headerBuffer.size()); // writing header
410 
411  if (!layout->HasBackupHeader())
412  {
413  return;
414  }
415 
416  // Write The Backup Header if any
417 
418  RandomGenerator::GetData(salt); // getting new salt
419  options.Hash->HMAC_DeriveKey(headerkey, *passwordkey, salt);
420  options.HeaderKey = headerkey;
421  header->Create(headerBuffer, options); // creating new header
422 
423  if (layout->GetBackupHeaderOffset() >= 0)
424  {
425  file.seekp(layout->GetBackupHeaderOffset(), std::ios_base::beg);
426  }
427  else
428  {
429  file.seekp(containerSize + layout->GetBackupHeaderOffset(), std::ios_base::beg);
430  }
431  file.write((char*)headerBuffer.get(),
432  headerBuffer.size()); // writing backup header crypted with new salt
