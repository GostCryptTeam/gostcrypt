47 static void TestCipher(CipherAlgorithm& cipher, const CipherTestVector* testVector,
48  size_t testVectorCount)
49 {
50  Buffer buffer(cipher.GetBlockSize());
51  for (size_t i = 0; i < testVectorCount; ++i)
52  {
53  cipher.SetKey(BufferPtr(testVector[i].Key, sizeof(testVector[i].Key)));
54  buffer.copyFrom(BufferPtr(testVector[i].Plaintext, buffer.size()));
55  cipher.EncryptBlock(buffer);
56 
57  if (memcmp(buffer, testVector[i].Ciphertext, buffer.size()) != 0)
58  {
59  throw EncryptionTestFailedException();
60  }
61  }
