116 void derive_u_stribog(char* pwd, int pwd_len, char* salt, int salt_len, int iterations, char* u,
117  int b)
118 {
119  char j[STRIBOG_DIGESTSIZE], k[STRIBOG_DIGESTSIZE];
120  char init[128];
121  char counter[4];
122  int c, i;
123 
124  /* iteration 1 */
125  memset(counter, 0, 4);
126  counter[3] = (char) b;
127  memcpy(init, salt, salt_len); /* salt */
128  memcpy(&init[salt_len], counter, 4); /* big-endian block number */
129  hmac_stribog(pwd, pwd_len, init, salt_len + 4, j);
130  memcpy(u, j, STRIBOG_DIGESTSIZE);
131 
132  /* remaining iterations */
133  for (c = 1; c < iterations; c++)
134  {
135  hmac_stribog(pwd, pwd_len, j, STRIBOG_DIGESTSIZE, k);
136  for (i = 0; i < STRIBOG_DIGESTSIZE; i++)
137  {
138  u[i] ^= k[i];
139  j[i] = k[i];
140  }
141  }
142 
143  /* Prevent possible leaks. */
144  burn(j, sizeof(j));
145  burn(k, sizeof(k));
146 }
