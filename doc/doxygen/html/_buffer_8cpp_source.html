75 void Buffer::allocate(size_t size)
76 {
77  if (size < 1)
78  {
79  throw InvalidParameterException("size", "size can not be null");
80  }
81 
82  Usersize = size;
83 
84  if (size <= Data.size())
85  {
86  return; // everything is fine and alloced already
87  }
88 
89  if (Data.size() != 0)
90  {
91  freeData();
92  }
93 
94  Usersize = size;
95

97  {
98  Data.set(static_cast<quint8*>(malloc(size)), size); // TODO : calloc ? Why not realloc ?

102  }
103  }
104  catch (...)
105  {

107  Usersize = 0;
