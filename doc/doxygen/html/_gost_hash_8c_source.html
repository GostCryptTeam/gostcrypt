39 static quint32 r(quint32 n1, quint32 n2, quint32 X)
40 {
41  gosthash_s_box* sbox = &GostR3411_94_CryptoProParamSet;
42  n1 += X;
43  n1 = (quint32)sbox->k8[(n1 >> 28) & 0xF] << 28 | (quint32)sbox->k7[(n1 >> 24) & 0xF] << 24
44  | (quint32)sbox->k6[(n1 >> 20) & 0xF] << 20 | (quint32)sbox->k5[(n1 >> 16) & 0xF] << 16
45  | (quint32)sbox->k4[(n1 >> 12) & 0xF] << 12 | (quint32)sbox->k3[(n1 >> 8) & 0xF] << 8
46  | (quint32)sbox->k2[(n1 >> 4) & 0xF] << 4 | (quint32)sbox->k1[n1 & 0xF];
47  n1 = gosthash_rotl32(n1, 11);
48  n2 ^= n1;
49  return n2;
50 }

167 static void add_blocks(quint8* T, quint8* F, qint32 len)
168 {
169  qint32 i;
170  qint16 carry = 0;
171  for (i = 0; i < len; i++)
172  {
173  qint16 sum = (qint16)T[i] + (qint16)F[i] + carry;
174  T[i] = (quint8)sum & 0xFF;
175  carry = sum >> 8;
176  }

188 /* GOST R 34.11-94 mixing functions */
189 /* This function maps bytes from F to T in the following way:
190  * T[0+0] = F[0+0]; T[1+0] = F[8+0]; ...; T[3+0] = F[24+0];
191  * T[0+4] = F[0+1]; T[1+4] = F[8+1]; ...;
192  * ...; T[2+28] = F[16+7]; T[3+28] = F[24+7];
193  */
194 static void P_transform(quint8* F, quint8* T)
195 {
196  qint32 i, k;
197  for (i = 0; i < 4; i++)
198  for (k = 0; k < 8; k++)
199  {
200  T[i + 4 * k] = F[8 * i + k];
201  }
202 }
203 
204 /* This function maps bytes from F to T in the following way:
205  * T[ 0..23] = F[8..31];
206  * T[24..31] = F[0..7] XOR F[8..15] */
207 static void A(quint8* F, quint8* T)
208 {
209  quint8 i;
210  for (i = 0; i < 24; i++)
211  {
212  T[i] = F[i + 8];
213  }
214  for (i = 0; i < 8; i++)
215  {
216  T[i + 24] = F[i] ^ F[i + 8];
217  }
218 }

220 static void PSI(quint8* X)
221 {
222  qint16 T;
223 
224  T = (qint16)(X[0] ^ X[2] ^ X[4] ^ X[6] ^ X[24] ^ X[30]) |
225  ((qint16)(X[1] ^ X[3] ^ X[5] ^ X[7] ^ X[25] ^ X[31]) << 8);
226  move_blocks(X, X + 2, 30);
227  X[30] = (quint8)(T & 0xFF);
228  X[31] = (quint8)(T >> 8);
229 }

236 static void step(quint8* H, quint8* M, gost_hash_ctx* ctx)
237 {
238  quint8 U[32], W[32], V[32], S[32], K[4][32];
239  qint32 i;
240 
241  (void)ctx;
242  xor_blocks(W, H, M, 32);
243  P_transform(W, K[0]); //First key
244 
245  gost_encrypt_with_key(H, S, K[0]);
246 
247  A(H, U);
248  A(M, V);
249  A(V, V);
250  xor_blocks(W, U, V, 32);
251  P_transform(W, K[1]); //Second key
252 
253  gost_encrypt_with_key(H + 8, S + 8, K[1]);
254 
255  A(U, U);
256  /* As the other C values are all 0's, we only need to XOR
257  * with C[3] */
258  xor_blocks(U, U, C_3, 32);
259  A(V, V);
260  A(V, V);
261  xor_blocks(W, U, V, 32);
262  P_transform(W, K[2]); //Third key
263 
264  gost_encrypt_with_key(H + 16, S + 16, K[2]);
265 
266  A(U, U);
267  A(V, V);
268  A(V, V);
269  xor_blocks(W, U, V, 32);
270  P_transform(W, K[3]); //Fourth key
271 
272  gost_encrypt_with_key(H + 24, S + 24, K[3]);
273 
274  for (i = 0; i < 12; i++)
275  {
276  PSI(S);
277  }
278  xor_blocks(S, S, M, 32);
279  PSI(S);
280  xor_blocks(S, S, H, 32);
281  for (i = 0; i < 61; i++)
282  {
283  PSI(S);
284  }
285  copy_blocks(H, S, 32);
286 }

288 void GOSTHASH_add(quint8* block, quint32 len, gost_hash_ctx* ctx)
289 {
290  quint8* curptr = block;
291  quint8* barrier = block + (len - 32); //In order that curptr += 32 won't overshoot len.
292 
293  if (ctx->left) //There are unsigned chars left from the last GOSTHASH_add
294  {
295  quint32 add_bytes = (32 - ctx->left) > len ? len : (32 - ctx->left);
296  copy_blocks(ctx->remainder + (quint8)ctx->left, block, (qint32)add_bytes);
297  if ((ctx->left + add_bytes) < 32) //This can be finished in the finalize stage if needed
298  {
299  return;
300  }
301 
302  curptr += add_bytes;
303  step(ctx->H, ctx->remainder, ctx);
304  add_blocks(ctx->S, ctx->remainder, 32);
305  ctx->len += 32;
306  ctx->left = 0;
307  }
308  else if (ctx->left + len < 32)
309  {
310  copy_blocks(ctx->remainder + (quint8)ctx->left, block, (qint32)len);
311  ctx->left += len;
312  return;
313  }
314 
315  while (curptr <= barrier) //Add the input block to the hash
316  {
317  step(ctx->H, curptr, ctx);
318 
319  add_blocks(ctx->S, curptr, 32); //Add 32 unsigned chars of the message to the encrypted message
320 
321  ctx->len += 32; //We've processed 32 unsigned chars
322  curptr += 32; //Advance the block-pointer 32 unsigned chars
323  }
324 
325  if (curptr != block +
326  len) //If we have unsigned chars remaining, add them for the next GOSTHASH_add of _finalize
327  {
328  ctx->left = (quint32)(block + len - curptr);
329  copy_blocks(ctx->remainder, curptr, (qint32)ctx->left);
330  }
331 }

333 void GOSTHASH_finalize(gost_hash_ctx* ctx, quint8* out)
334 {
335  quint8 buf[32];
336  quint8 H[32];
337  quint8 S[32];
338  qint32 final_len;
339  qint32 bptr;
340 
341  final_len = ctx->len;
342 
343  copy_blocks(H, ctx->H, 32);
344  copy_blocks(S, ctx->S, 32);
345 
346  if (ctx->left) //Handle any remaining bytes
347  {
348  set_blocks(buf, 0, 32);
349  copy_blocks(buf, ctx->remainder, (qint32)ctx->left);
350  step(H, buf, ctx);
351  add_blocks(S, buf, 32);
352  final_len += ctx->left;
353  }
354 
355  set_blocks(buf, 0, 32);
356  bptr = 0;
357  final_len <<= 3;
358  while (final_len > 0)
359  {
360  buf[bptr++] = (quint8)(final_len & 0xFF);
361  final_len >>= 8;
362  }
363 
364  step(H, buf, ctx);
365  step(H, S, ctx);
366  copy_blocks(out, H, 32);
367 }
