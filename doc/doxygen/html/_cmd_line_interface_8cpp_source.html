233 
234  termios oldt;
235  tcgetattr(STDIN_FILENO, &oldt);
236  termios newt = oldt;
237  newt.c_lflag &= ~ECHO;
238  tcsetattr(STDIN_FILENO, TCSANOW, &newt); // hide input
239 
240  qStdOut() << "Please enter your sudo password: " << endl;
241 
242  // Temporaire mot de passe toujours en mémoire
243  QFile inputFile;

245  inputFile.open(stdin, QFile::ReadOnly);

247  inputFile.close();
248  *password = line.toUtf8();
249  line.fill('\0');
