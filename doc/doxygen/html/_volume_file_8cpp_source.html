76 void VolumeFile::Open(const QFileInfo path, bool readOnly, bool preserveTimestamps)
77 {
78  int sysFlags = O_LARGEFILE;
79 
80 
81  if (readOnly)
82  {
83  sysFlags |= O_RDONLY;
84  }
85  else
86  {
87  sysFlags |= O_RDWR;
88  }
89 
90  if (preserveTimestamps)
91  {
92  struct stat statData;
93  if (stat(path.absoluteFilePath().toLocal8Bit().data(), &statData) == -1)
94  {
95  throw FailedGetTimestampsException();
96  }
97  AccTime = statData.st_atime;
98  ModTime = statData.st_mtime;
99  }
100 
101  FileHandle = open(path.absoluteFilePath().toLocal8Bit().data(), sysFlags, S_IRUSR | S_IWUSR);
102  if (FileHandle == -1)
103  {
104  throw FailedOpenFileException(Path);
105  }
106 
107  Path = path;
108  this->preserveTimestamps = preserveTimestamps;
109  FileIsOpen = true;
110 }

197 FileType::Enum VolumeFile::GetType() const
198 {
199  struct stat statData;
200  if (stat(Path.absoluteFilePath().toLocal8Bit().data(), &statData) != 0)
201  {
202  throw FailedStatFileException(Path);
203  }
204 
205  if (S_ISREG(statData.st_mode)) { return FileType::File; }
206  if (S_ISCHR(statData.st_mode)) { return FileType::CharacterDevice; }
207  if (S_ISBLK(statData.st_mode)) { return FileType::BlockDevice; }
208  if (S_ISLNK(statData.st_mode)) { return FileType::SymbolickLink; }
209 
210  return FileType::Unknown;
211 }
